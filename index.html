    function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('待辦');
  if (!sheet) sheet = ss.insertSheet('待辦');
  var data = JSON.parse(e.postData.contents);
  var tasks = data.tasks || [];

  sheet.clear();
  sheet.appendRow(['ID','內容','日期','狀態','同步時間']);

  for (var i = 0; i < tasks.length; i++) {
    var t = tasks[i];
    sheet.appendRow([
      t.id,
      t.text,
      t.date || '今日',
      t.done ? '✅ 完成' : '⬜ 未完成',
      new Date().toLocaleString('zh-TW')
    ]);
  }

  return ContentService
    .createTextOutput(
      JSON.stringify({ok:true, count:tasks.length})
    )
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService
    .createTextOutput(
      JSON.stringify({ok:true, msg:'連線成功'})
    )
    .setMimeType(ContentService.MimeType.JSON);
}
